## 节流：
一定时间内只首次触发生效

``
function throttle (fn, timeout) {
    let timer = null;
    return function (...args) {
        if (timer) {
            return;
        }
        const _this = this;
        timer = setTimeout(function () {
            fn.apply(_this, args);
            timer = null;
        }, timeout);
    }
}
``

## 防抖：
一定时间内只最后一次触发生效

``
function debounce (fn, timeout) {
    let timer = null;
    return function (...args) {
        if (timer) {
            clearTimeout(timer);
        }
        const _this = this;
        timer = setTimeout(function () {
            fn.apply(_this, args);
        }, timeout);
    }
}
``
